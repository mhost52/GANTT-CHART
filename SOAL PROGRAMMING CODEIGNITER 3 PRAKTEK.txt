## 📂 Model (`Mahasiswa_model.php`)
```php
<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class Mahasiswa_model extends CI_Model {

    // Ambil semua data mahasiswa dengan join jurusan & dosen
    public function get_all() {
        $this->db->select('mahasiswa.id_mahasiswa, mahasiswa.nama_mahasiswa, mahasiswa.nim, jurusan.nama_jurusan, dosen.nama_dosen');
        $this->db->from('mahasiswa');
        $this->db->join('jurusan', 'jurusan.id_jurusan = mahasiswa.id_jurusan');
        $this->db->join('dosen', 'dosen.id_dosen = mahasiswa.id_dosen');
        return $this->db->get()->result();
    }

    // Tambah data mahasiswa
    public function insert($data) {
        return $this->db->insert('mahasiswa', $data);
    }

    // Ambil data mahasiswa berdasarkan id
    public function get_by_id($id) {
        return $this->db->get_where('mahasiswa', ['id_mahasiswa' => $id])->row();
    }

    // Update data mahasiswa
    public function update($id, $data) {
        $this->db->where('id_mahasiswa', $id);
        return $this->db->update('mahasiswa', $data);
    }

    // Hapus data mahasiswa
    public function delete($id) {
        return $this->db->delete('mahasiswa', ['id_mahasiswa' => $id]);
    }
}
```

## 📂 Controller (`Mahasiswa.php`)
```php
<?php
defined('BASEPATH') OR exit('No direct script access allowed');

class Mahasiswa extends CI_Controller {

    public function __construct() {
        parent::__construct();
        $this->load->model('Mahasiswa_model');
        $this->load->library('form_validation');
    }

    // Tampilkan daftar mahasiswa
    public function index() {
        $data['mahasiswa'] = $this->Mahasiswa_model->get_all();
        $this->load->view('mahasiswa_list', $data);
    }

    // Tambah mahasiswa
    public function create() {
        $this->form_validation->set_rules('nim', 'NIM', 'required|is_unique[mahasiswa.nim]');
        $this->form_validation->set_rules('nama_mahasiswa', 'Nama', 'required');

        if ($this->form_validation->run() == FALSE) {
            $this->load->view('mahasiswa_form');
        } else {
            $data = [
                'nim' => $this->input->post('nim'),
                'nama_mahasiswa' => $this->input->post('nama_mahasiswa'),
                'id_jurusan' => $this->input->post('id_jurusan'),
                'id_dosen' => $this->input->post('id_dosen')
            ];
            $this->Mahasiswa_model->insert($data);
            redirect('mahasiswa');
        }
    }

    // Edit mahasiswa
    public function edit($id) {
        $data['mahasiswa'] = $this->Mahasiswa_model->get_by_id($id);

        if ($this->input->post()) {
            $update = [
                'nim' => $this->input->post('nim'),
                'nama_mahasiswa' => $this->input->post('nama_mahasiswa'),
                'id_jurusan' => $this->input->post('id_jurusan'),
                'id_dosen' => $this->input->post('id_dosen')
            ];
            $this->Mahasiswa_model->update($id, $update);
            redirect('mahasiswa');
        }

        $this->load->view('mahasiswa_form', $data);
    }

    // Hapus mahasiswa
    public function delete($id) {
        $this->Mahasiswa_model->delete($id);
        redirect('mahasiswa');
    }
}
```
